使用正規化

mysql> SELECT v
-> FROM
-> (
-> SELECT ‘sql server是微软开发的数据库管理系统’ AS v UNION ALL
-> SELECT ‘C-123456789’ AS v UNION ALL
-> SELECT ‘MySQL是一个开源的数据库管理系统’ AS v UNION ALL
-> SELECT ‘sdalfkj’
-> )t
-> WHERE v REGEXP ‘[吖-座]’=1;
